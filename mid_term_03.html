ggplot2 繪圖練習

  • 散布圖 線圖 直方圖 盒鬚圖 長條圖

    • 散布圖
    library(ggplot2)
data(msleep)
ggplot.1 <- ggplot(data=msleep,aes(x=bodywt,y=brainwt,color=factor(vore)))
lm1 <- lm(brainwt~bodywt,data=msleep)
ggplot.1+
  geom_point()+
  xlim(c(0,250))+
  ylim(c(0,2))+
  geom_abline(slope=lm1$coef)

    • 線圖
    data("txhousing")
Date <- rep(seq(as.Date("2000/01/01"),by="month",length=187),46)
txhousing.Date <- cbind(txhousing,Date)
ggplot.2 <- ggplot(txhousing.Date,aes(x=Date,y=median))
ggplot.2+
  geom_line()+
  facet_wrap(~city)
